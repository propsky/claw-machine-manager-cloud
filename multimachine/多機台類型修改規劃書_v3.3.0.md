## 🔧 Backend API Changes Required

### 1️⃣ Database Schema

```sql
-- Machine Types Table
CREATE TABLE machine_types (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(50) NOT NULL,       -- 類型名稱
  code VARCHAR(20) UNIQUE NOT NULL, -- 代碼
  icon VARCHAR(10),                -- 圖示
  coin_price INT,                  -- 單價
  has_inventory BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Insert Default Data
INSERT INTO machine_types (name, code, icon, coin_price, has_inventory) VALUES
('娃娃機', 'claw', '🧸', 30, FALSE),
('扭蛋機', 'gacha', '🥚', 30, TRUE),
('打地鼠', 'whack', '🐹', 30, FALSE),
('搖馬機', 'rocking', '🐴', 30, FALSE),
('販賣機', 'vending', '🥤', NULL, TRUE);

-- Add column to existing table
ALTER TABLE happy_cardmachines 
ADD COLUMN machine_type VARCHAR(20) DEFAULT 'claw' 
COMMENT 'claw|gacha|whack|rocking|vending';

-- Add inventory column
ALTER TABLE happy_cardmachines 
ADD COLUMN stock_quantity INT DEFAULT 0 
COMMENT '庫存數量';
```
